name: Build & Release

on:
  push:
    tags:
      - "v*"

jobs:
  build:
    name: Build and Upload Release
    strategy:
      matrix:
        include:
          - os: ubuntu-latest
            target: x86_64-unknown-linux-gnu
            bin: auto_update_app
          - os: windows-latest
            target: x86_64-pc-windows-msvc
            bin: auto_update_app.exe
          - os: macos-latest
            target: x86_64-apple-darwin
            bin: auto_update_app

    runs-on: ${{ matrix.os }}

    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Install Rust
        uses: dtolnay/rust-toolchain@stable

      - name: Build
        run: cargo build --release

      - name: Prepare binary
        shell: bash
        run: |
          mkdir dist
          cp target/release/${{ matrix.bin }} dist/auto_update_app-${{ matrix.target }}

      - name: Generate checksum
        shell: bash
        run: |
          cd dist
          sha256sum * > checksums.txt

      - name: Upload Release
        uses: softprops/action-gh-release@v2
        with:
          files: dist/*
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
